159  void actionsAfterTimeStep() override
160  {
161  indenter_.sumForceTorqueOMP();
162 
163  static Mdouble timeToRetract = 0;
164  indenter_.move(indenter_.getVelocity() * getTimeStep());
165 
166  if (getIndenterVelocity() < 0)
167  { //moving down
168  if (getForceOnIndenter() > 0 && getIndenterVelocity()!=-indentationVelocity_)
169  {
170  setIndenterVelocity(-indentationVelocity_);
171  logger(INFO, "Lowering indentation velocity %", getIndenterVelocity());
172  }
173  if (getForceOnIndenter() >= indentationForce_)
174  {
175  setIndenterVelocity(0);
176  timeToRetract = 1.1 * getTime();
177  logger(INFO, "stopping indenter");
178  }
179  }
180  else if (getIndenterVelocity() == 0)
181  { //stopping
182  if (getTime()>timeToRetract) {
183  setIndenterVelocity(indentationVelocity_);
184  logger(INFO, "retracting indenter");
185  measuredIndentationForce = getForceOnIndenter();
186  measuredIndentation = getIndenterHeight();
187  }
188  } else
189  { //retracting
190  if (measuredForceGradient==0 && getForceOnIndenter() < 0.5*measuredIndentationForce)
191  {
192  measuredForceGradient = (0.5 * measuredIndentationForce) / (getIndenterHeight() - measuredIndentation);
193  logger(INFO, "measured force gradient = %", measuredForceGradient);
194  } else if (getForceOnIndenter() > 0)
195  {
196  //logger(INFO, "stopping indenter");
197  setTimeMax(getTime() * 1.2);
198  }
199  else if (measuredElasticDisplacement == 0)
200  {
201  measuredElasticDisplacement = getIndenterHeight() - measuredIndentation;
202  logger(INFO, "measured elastic displacement = %\n"
203  "measured Elasticity = %", measuredElasticDisplacement, Flusher::NO_FLUSH,
204  0.5 * measuredForceGradient / sqrt(0.5 * indenterDiameter_ * measuredElasticDisplacement));
205  setIndenterVelocity(10.0 * indentationVelocity_);
206  logger(INFO, "Increasing indentation velocity %", getIndenterVelocity());
207  }
208  }
209  }
210 
212  Mdouble getBedHeight()
213  {
214  Mdouble bedHeight = getZMin();
215  for (auto p : particleHandler)
216  if (!p->isFixed())
217  bedHeight = std::max(bedHeight, p->getPosition().Z + p->getRadius());
218  logger(INFO, "bed height % N=%", bedHeight, particleHandler.getNumberOfObjects());
219  //return 1;
220  return bedHeight;
221  }
