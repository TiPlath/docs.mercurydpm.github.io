125  const Vector<std::complex<double>>& rhs)
126  {
127  unsigned long n = rhs.size();
128  Vector<std::complex<double>> res(n);
129  residual(x, rhs, res);
130  double ans = 0.0;
131  for (unsigned long i = 0; i < n; i++)
132  {
133  ans = std::max(ans, std::abs(res[i]));
134  }
135  return ans;
136  }
