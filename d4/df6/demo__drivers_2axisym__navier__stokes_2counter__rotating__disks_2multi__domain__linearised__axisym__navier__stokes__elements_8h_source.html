51 class LinearisedAxisymmetricQTaylorHoodMultiDomainElement :
52 public virtual LinearisedAxisymmetricQTaylorHoodElement,

58  LinearisedAxisymmetricQTaylorHoodMultiDomainElement() :
59  LinearisedAxisymmetricQTaylorHoodElement(),
60  ElementWithExternalElement()
61  {
62  // There are two interactions: the base flow velocities and their
63  // derivatives w.r.t. global coordinates
64  this->set_ninteraction(2);
65 
66  // Do not include any external interaction data when computing
67  // the element's Jacobian
68  ElementWithExternalElement::ignore_external_interaction_data();
69 
72  ElementWithExternalElement::ignore_external_geometric_data();

77  virtual void get_base_flow_u(const double& time,
78  const unsigned& ipt,
79  const Vector<double>& x,
80  Vector<double>& result) const
81  {
82  // Set interaction index to 0
83  const unsigned interaction = 0;
84 
85  // Get a pointer to the external element that computes the base flow.
86  // We know that it's an axisymmetric Navier-Stokes element.
87  const AxisymmetricQTaylorHoodElement* base_flow_el_pt =
88  dynamic_cast<AxisymmetricQTaylorHoodElement*>(
89  external_element_pt(interaction,ipt));
90 
91  // Provide storage for local coordinates in the external element
92  // which correspond to the integration point ipt
93  Vector<double> s_external(2);
94 
95  // Determine local coordinates in the external element which correspond
96  // to the integration point ipt
97  s_external = external_element_local_coord(interaction,ipt);
98 
99  // Get the three velocity components interpolated from the external element
100  for(unsigned i=0;i<3;i++)
101  {
102  result[i] = base_flow_el_pt->interpolated_u_axi_nst(s_external,i);
103  }
104 
105  } // End of overloaded get_base_flow_u function
