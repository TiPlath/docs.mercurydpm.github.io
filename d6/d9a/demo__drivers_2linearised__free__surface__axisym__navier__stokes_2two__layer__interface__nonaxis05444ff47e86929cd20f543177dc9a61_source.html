79  virtual void get_base_flow_u(const double& time,
80  const unsigned& ipt,
81  const Vector<double>& x,
82  Vector<double>& result) const
83  {
84  // Set interaction index to 0
85  const unsigned interaction = 0;
86 
87  // Get a pointer to the external element that computes the base flow.
88  // We know that it's an axisymmetric Navier-Stokes element.
89  const AxisymmetricQCrouzeixRaviartElement* base_flow_el_pt =
90  dynamic_cast<AxisymmetricQCrouzeixRaviartElement*>(
91  external_element_pt(interaction,ipt));
92 
93  // Provide storage for local coordinates in the external element
94  // which correspond to the integration point ipt
95  Vector<double> s_external(2);
96 
97  // Determine local coordinates in the external element which correspond
98  // to the integration point ipt
99  s_external = external_element_local_coord(interaction,ipt);
100 
101  // Get the three velocity components interpolated from the external element
102  for(unsigned i=0;i<3;i++)
103  {
104  result[i] = base_flow_el_pt->interpolated_u_axi_nst(s_external,i);
105  }
106 
107  } // End of overloaded get_base_flow_u function
