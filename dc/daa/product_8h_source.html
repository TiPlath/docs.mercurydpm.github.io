13 template <typename Derived1, typename Derived2>
14 bool areNotApprox(const MatrixBase<Derived1>& m1, const MatrixBase<Derived2>& m2,
15  typename Derived1::RealScalar epsilon = NumTraits<typename Derived1::RealScalar>::dummy_precision()) {
16  return !((m1 - m2).cwiseAbs2().maxCoeff() <
17  epsilon * epsilon * (std::max)(m1.cwiseAbs2().maxCoeff(), m2.cwiseAbs2().maxCoeff()));
18 }
