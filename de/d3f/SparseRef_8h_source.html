81  public:

86  : Base(RowsAtCompileTime == Dynamic ? 0 : RowsAtCompileTime, ColsAtCompileTime == Dynamic ? 0 : ColsAtCompileTime,
87  0, 0, 0, 0, 0) {}
88 
89  protected:
90  template <typename Expression>

92  if (expr.outerIndexPtr() == 0)

94  else
