104  EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device) : m_impl(op.expression(), device) {
105  m_dimensions = m_impl.dimensions();
106  for (int i = 0; i < NumDims; ++i) {
107  m_dimensions[i] = Eigen::numext::ceil(static_cast<float>(m_dimensions[i]) / op.strides()[i]);
108  }
109 
110  const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
111  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
112  m_outputStrides[0] = 1;
113  m_inputStrides[0] = 1;
114  for (int i = 1; i < NumDims; ++i) {
115  m_outputStrides[i] = m_outputStrides[i - 1] * m_dimensions[i - 1];
116  m_inputStrides[i] = m_inputStrides[i - 1] * input_dims[i - 1];
117  m_inputStrides[i - 1] *= op.strides()[i - 1];
118  }
119  m_inputStrides[NumDims - 1] *= op.strides()[NumDims - 1];
120  } else { // RowMajor
121  m_outputStrides[NumDims - 1] = 1;
122  m_inputStrides[NumDims - 1] = 1;
123  for (int i = NumDims - 2; i >= 0; --i) {
124  m_outputStrides[i] = m_outputStrides[i + 1] * m_dimensions[i + 1];
125  m_inputStrides[i] = m_inputStrides[i + 1] * input_dims[i + 1];
126  m_inputStrides[i + 1] *= op.strides()[i + 1];
127  }
128  m_inputStrides[0] *= op.strides()[0];
129  }
130  }

144  template <int LoadMode>
145  EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const {
146  EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
147  eigen_assert(index + PacketSize - 1 < dimensions().TotalSize());
148 
149  Index inputIndices[] = {0, 0};
150  Index indices[] = {index, index + PacketSize - 1};
151  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
152  EIGEN_UNROLL_LOOP
153  for (int i = NumDims - 1; i > 0; --i) {
154  const Index idx0 = indices[0] / m_outputStrides[i];
155  const Index idx1 = indices[1] / m_outputStrides[i];
156  inputIndices[0] += idx0 * m_inputStrides[i];
157  inputIndices[1] += idx1 * m_inputStrides[i];
158  indices[0] -= idx0 * m_outputStrides[i];
159  indices[1] -= idx1 * m_outputStrides[i];
160  }
161  inputIndices[0] += indices[0] * m_inputStrides[0];
162  inputIndices[1] += indices[1] * m_inputStrides[0];
163  } else { // RowMajor
164  EIGEN_UNROLL_LOOP
165  for (int i = 0; i < NumDims - 1; ++i) {
166  const Index idx0 = indices[0] / m_outputStrides[i];
167  const Index idx1 = indices[1] / m_outputStrides[i];
168  inputIndices[0] += idx0 * m_inputStrides[i];
169  inputIndices[1] += idx1 * m_inputStrides[i];
170  indices[0] -= idx0 * m_outputStrides[i];
171  indices[1] -= idx1 * m_outputStrides[i];
172  }
173  inputIndices[0] += indices[0] * m_inputStrides[NumDims - 1];
174  inputIndices[1] += indices[1] * m_inputStrides[NumDims - 1];
175  }
176  if (inputIndices[1] - inputIndices[0] == PacketSize - 1) {
177  PacketReturnType rslt = m_impl.template packet<Unaligned>(inputIndices[0]);
178  return rslt;
179  } else {
180  EIGEN_ALIGN_MAX std::remove_const_t<CoeffReturnType> values[PacketSize];
181  values[0] = m_impl.coeff(inputIndices[0]);
182  values[PacketSize - 1] = m_impl.coeff(inputIndices[1]);
183  EIGEN_UNROLL_LOOP
184  for (int i = 1; i < PacketSize - 1; ++i) {
185  values[i] = coeff(index + i);
186  }
187  PacketReturnType rslt = internal::pload<PacketReturnType>(values);
188  return rslt;
189  }
190  }

192  EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost costPerCoeff(bool vectorized) const {
193  double compute_cost = (NumDims - 1) * (TensorOpCost::AddCost<Index>() + TensorOpCost::MulCost<Index>() +
194  TensorOpCost::DivCost<Index>()) +
195  TensorOpCost::MulCost<Index>();
196  if (vectorized) {
197  compute_cost *= 2; // packet() computes two indices
198  }
199  const int innerDim = (static_cast<int>(Layout) == static_cast<int>(ColMajor)) ? 0 : (NumDims - 1);
200  return m_impl.costPerCoeff(vectorized && m_inputStrides[innerDim] == 1) +
201  // Computation is not vectorized per se, but it is done once per packet.
202  TensorOpCost(0, 0, compute_cost, vectorized, PacketSize);
203  }

208  EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index srcCoeff(Index index) const {
209  Index inputIndex = 0;
210  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
211  EIGEN_UNROLL_LOOP
212  for (int i = NumDims - 1; i > 0; --i) {
213  const Index idx = index / m_outputStrides[i];
214  inputIndex += idx * m_inputStrides[i];
215  index -= idx * m_outputStrides[i];
216  }
217  inputIndex += index * m_inputStrides[0];
218  } else { // RowMajor
219  EIGEN_UNROLL_LOOP
220  for (int i = 0; i < NumDims - 1; ++i) {
221  const Index idx = index / m_outputStrides[i];
222  inputIndex += idx * m_inputStrides[i];
223  index -= idx * m_outputStrides[i];
224  }
225  inputIndex += index * m_inputStrides[NumDims - 1];
226  }
227  return inputIndex;
228  }
229 
230  Dimensions m_dimensions;
231  array<Index, NumDims> m_outputStrides;
232  array<Index, NumDims> m_inputStrides;
233  TensorEvaluator<ArgType, Device> m_impl;
234 };
